%\renewcommand*\thesection{\Alph{section}} % change section numbering to Alphabet
% Unnumbered subsection for the nice subsection titles and entry into TOC + pdfBookmark	
\providecommand*\phantomsection{}	% if hyperref package missing, don't error out

\providecommand*\usubsection[1]{	% unnembered subection command with 1 input
	\subsection*{#1}				% original subsection without numbering and no TOC
	\phantomsection					% for pdfBookmarks links to point correctly (only
									% nessicary when dealing with unnumbered sections)
	\addcontentsline{toc}{subsection}{#1}} % add entry to TOC using title

\providecommand*\usubsubsection[1]{	% unnembered subection command with 1 input
	\subsubsection*{#1}				% original subsection without numbering and no TOC
	\phantomsection					% for pdfBookmarks links to point correctly (only
									% nessicary when dealing with unnumbered sections)
	\addcontentsline{toc}{subsubsection}{#1}} % add entry to TOC using title		

% For \url{SOME_URL}, links SOME_URL to the url SOME_URL
\providecommand*\url[1]{\href{#1}{#1}}
% Same as above, but pretty-prints SOME_URL in teletype fixed-width font
\renewcommand*\url[1]{\href{#1}{\texttt{#1}}}

% For \email{ADDRESS}, links ADDRESS to the url mailto:ADDRESS
\providecommand*\email[1]{\href{mailto:#1}{#1}}
% Same as above, but pretty-prints ADDRESS in teletype fixed-width font
\renewcommand*\email[1]{\href{mailto:#1}{\texttt{#1}}}

% Caption formating for code
\DeclareCaptionFont{white}{\color{white}}

	{\colorbox{gray}{\parbox{\dimexpr\textwidth-2\fboxsep\relax}{#1#2#3}}}
\captionsetup[lstlisting]{format=listing,labelfont=white,textfont=white}
